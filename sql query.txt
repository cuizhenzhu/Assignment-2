FROM Member m

(c)
   
SELECT t1.MemberID, t2.LastName, t2.FirstName
FROM (
 SELECT DISTINCT MemberID
 FROM Entry
 WHERE year IN ('2014', '2015')
) t1
 LEFT JOIN (
  SELECT MemberID, LastName, FirstName
  FROM Member
 ) t2
 ON t1.MemberID = t2.MemberID;


(d)

SELECT e1.MemberID
FROM Entry e1, Entry e2, Entry e3
WHERE e1.MemberID = e2.MemberID and 
e1.MemberID = e3.MemberID and e3.MemberID = e2.MemberID and
e1.Year= 2013 and e2.Year = 2014 and e3.Year= 2015
GROUP BY e1.MemberID;
